const 錯誤訊息 = require('./ErrorMesages')

function format (template) {
    if (arguments.length < 2) {
        throw new Error(錯誤訊息.format.至少要有二個參數)
    }

    if (!template) {
        throw new Error(錯誤訊息.format.模板不可以是空值)
    }

    let variables = template.match(/{([0-9]+)}/gi)
    if (!variables) {
        throw new Error(錯誤訊息.format.模板裡面未設定變數)
    }

    if (variables.length !== arguments.length - 1) {
        throw new Error(錯誤訊息.format.參數與模板裡面的變數數量不同)
    }

    let result = template
    for (let i = 1; i < arguments.length; i++) {
        let regex = new RegExp('\\{' + i + '\\}', 'gi')
        if (!regex.test(template)) {
            throw new Error('在模板裡面沒有 ' + i + 1 + ' 的變數')
        }
        result = result.replace(regex, arguments[i])
    }

    return result
}

/**
 *
 * @param pattern
 * @param count
 * @returns {string}
 */
function repeat (pattern, count) {
    if (arguments.length < 2 || count < 0 || (!pattern)) return undefined
    return new Array(count + 1).join(pattern)
}

/**
 * 判斷findAll在句子中。
 * @param {*} 詞彙陣列 詞彙陣列，["new", "york"]
 * @param {string} 句子 句子。
 * @return {boolean}
 */
const findAll = (句子, 詞彙陣列) => {
    for (let 詞彙 = 0; 詞彙 < 詞彙陣列.length; 詞彙++) {
        if (句子.indexOf(詞彙陣列[詞彙]) === -1) return false
    }
    return true
}

/**
 * 移除空白字元(\r\t\n\f)。
 * @param {string} str
 * @returns {string}
 */
const 移除空白字元 = (str = '') => {
    return str.replace(/\s/g, '')
}
